function parseFile(file) {
  return new Promise(resolve => {
    const reader = new FileReader();
    reader.onload = e => {
      try {
        const data = new Uint8Array(e.target.result);
        const wb = XLSX.read(data, {type:'array'});
        const sheet = wb.Sheets[wb.SheetNames[0]];
        const rows = XLSX.utils.sheet_to_json(sheet, {header:1, defval:''});
        if (!rows.length) throw new Error('Empty file');
        const headers = rows[0].map(h => String(h).trim()).filter(h => h !== '');
        const dataRows = rows.slice(1).filter(r => r.some(c => c !== '' && c != null)).map(r => {
          const o = {}; headers.forEach((h,i) => o[h] = r[i] !== undefined ? r[i] : ''); return o;
        });
        const id = 'f_' + Date.now() + '_' + Math.random().toString(36).slice(2,7);
        const type = detectFileType(headers, file.name);
        files.push({ id, name:file.name, type, headers, rows:dataRows, mapping:autoMap(type, headers), elections:detectElections(headers, type) });
        resolve();
      } catch(err) { showToast('Error: ' + err.message); resolve(); }
    };
    reader.readAsArrayBuffer(file);
  });
}

function detectFileType(headers, name) {
  const upper = headers.map(h => h.toUpperCase().replace(/\s+/g,''));
  const hasStateId  = upper.includes('STATEID');
  const hasStateVID = upper.includes('STATEVOTERID') || upper.includes('VOTERKEY');
  const hasVHcols   = upper.some(h => /^VH\d{2}[A-Z]$/.test(h));
  const hasVoteType = headers.some(h => /- VOTE TYPE$/i.test(h));
  if ((hasStateVID && hasVHcols) || (hasStateVID && !hasVoteType)) return 'voter_enriched';
  if (hasStateId && hasVoteType) return 'voter_individual';
  if (hasVoteType) return 'voter_individual';
  const h = headers.join(' ').toLowerCase();
  if (h.includes('lat') && (h.includes('lng') || h.includes('lon')) && !h.includes('registered')) return 'geo';
  if (h.includes('demvote') || h.includes('d_vote')) return 'results';
  if (h.includes('registered')) return 'voter';
  return 'other';
}

function detectElections(headers, type) {
  const map = {};
  headers.forEach(h => {
    const u = h.toUpperCase().trim();
    // IL-style
    const vt = u.match(/^(.+?)\s*-\s*VOTE TYPE$/);
    const bc = u.match(/^(.+?)\s*-\s*BALLOT CHOICE$/);
    if (vt) { const k = vt[1].trim(); map[k] = map[k] || {name:k,style:'IL',voteTypeCol:null,ballotChoiceCol:null}; map[k].voteTypeCol = h; }
    if (bc) { const k = bc[1].trim(); map[k] = map[k] || {name:k,style:'IL',voteTypeCol:null,ballotChoiceCol:null}; map[k].ballotChoiceCol = h; }
    // VH-style: VH##L where L = G/P/M/Y/S/R/B/A  → VH20G = 2020 General
    const vh = u.match(/^VH(\d{2})([A-Z])$/);
    if (vh) {
      const yy = parseInt(vh[1],10);
      const year = yy >= 50 ? 1900+yy : 2000+yy;
      const letter = vh[2];
      const tMap = {G:'general',P:'primary',M:'consolidated',Y:'consolidated primary',S:'special',R:'runoff',B:'special',A:'special'};
      const typeLabel = tMap[letter] || 'other';
      const k = `VH${vh[1]}${letter}`;
      map[k] = {name:`${year} ${typeLabel}`, style:'VH', voteTypeCol:h, ballotChoiceCol:null, _year:year, _type:typeLabel};
    }
  });
  return Object.values(map).map(e => {
    if (e.style === 'VH') { e.year = e._year; e.type = e._type; delete e._year; delete e._type; return e; }
    const ym = e.name.match(/(20\d{2}|19\d{2})/);
    e.year = ym ? parseInt(ym[1]) : null;
    const U = e.name.toUpperCase();
    if (U.includes('GENERAL')) e.type = 'general';
    else if (U.includes('PRIMARY') || U.includes('PRIMARIA')) e.type = 'primary';
    else if (U.includes('CONSOLID') || U.includes('CNSLD')) e.type = 'consolidated';
    else e.type = 'other';
    return e;
  }).filter(e => e.year).sort((a,b) => b.year - a.year || (a.type==='general'?-1:1));
}

function autoMap(type, headers) {
  const m = {};
  (fieldDefs[type] || []).forEach(f => {
    const lower = headers.map(h => h.toLowerCase().replace(/\s+/g,'').trim());
    for (const syn of (f.syn || [])) {
      const s = syn.toLowerCase().replace(/\s+/g,'');
      for (let i = 0; i < lower.length; i++) {
        if (lower[i] === s) { m[f.key] = headers[i]; return; }
      }
      for (let i = 0; i < lower.length; i++) {
        if (lower[i].includes(s)) { m[f.key] = headers[i]; return; }
      }
    }
  });
  return m;
}

    <tbody>${f.rows.slice(0,3).map(r=>`<tr class="border-b">${previewHeaders.map(h=>`<td class="p-1">${String(r[h]).slice(0,20)}</td>`).join('')}${f.headers.length>12?'<td>...</td>':''}</tr>`).join('')}</tbody>`;
}

// ================ PROCESSING ================
function num(v) {
  if (v === '' || v == null) return null;
  if (typeof v === 'number') return v;
  const n = parseFloat(String(v).replace(/[,%$\s]/g,''));
  return isNaN(n) ? null : n;
}
function classifyBallot(val) {
  if (!val) return null;
  const s = String(val).toUpperCase().trim();
  if (!s) return null;
  if (s.startsWith('DEM') || s === 'D') return 'D';
  if (s.startsWith('REP') || s === 'R' || s === 'GOP') return 'R';
  if (s.startsWith('NONPART') || s.startsWith('NON-PART')) return 'N';
  if (s.startsWith('GRN') || s.startsWith('GREEN')) return 'G';
  if (s.startsWith('LIB')) return 'L';
  return 'O';
}
function normParty(v) {
  if (!v) return null;
  const s = String(v).toUpperCase().trim();
  if (!s || s === 'N' || s === 'NONE' || s === 'UNK') return null;
  if (s.startsWith('DEM') || s === 'D') return 'D';
  if (s.startsWith('REP') || s === 'R') return 'R';
  if (s.startsWith('UNA') || s === 'U' || s === 'I' || s.startsWith('IND') || s.startsWith('NON')) return 'U';
  return null;
}
function truthy(v) {
  if (v === null || v === undefined || v === '') return false;
  const s = String(v).toUpperCase().trim();
  return s === 'Y' || s === 'YES' || s === 'TRUE' || s === '1' || s === 'T';
}

// Build or update a voter from an IL file row
function buildVoterFromIL(r, f) {
  const m = f.mapping;
  const pid = m.precinct ? String(r[m.precinct] ?? '').trim() : '';
  const voterId = m.voterId ? String(r[m.voterId] ?? '').trim() : '';
  if (!voterId) return null;
  const age = m.age ? num(r[m.age]) : null;
  const firstName = m.firstName ? r[m.firstName] : '';
  const lastName  = m.lastName  ? r[m.lastName]  : '';
  const fullName = `${firstName||''} ${lastName||''}`.trim() || `(Voter ${voterId})`;
  const lat = m.lat ? num(r[m.lat]) : null;
  const lng = m.lng ? num(r[m.lng]) : null;
  const houseNum = m.houseNum ? r[m.houseNum] : '';
  const street = m.street ? r[m.street] : '';
  const city = m.city ? r[m.city] : '';
  const zip = m.zip ? r[m.zip] : '';
  const addr = [houseNum, street, city, zip].filter(x => x && String(x).trim()).join(' ');

  return {
    id: voterId, name: fullName, firstName: firstName||'', lastName: lastName||'',
    age: age, precinctId: pid, address: addr, city, zip, lat, lng,
    sources: { il: true, enriched: false },
    // Placeholders - filled later:
    phone:'', mobile:'', landline:'', dob:'', gender:'', suffix:'',
    observedParty:null, officialParty:null, calculatedParty:null,
    overallFreq:null, generalFreq:null, primaryFreq:null,
    cd:'', ld:'', sd:'', county:'', state:'',
    propensity:0, party:'U', partisanScore:0,
    dBallots:0, rBallots:0, otherBallots:0,
    votedGenerals:0, totalGenerals:0, votedPrimaries:0, totalPrimaries:0,
    last5Count:0, last5Voted:0,
    isSuperVoter:false, isBipartisanVoter:false,
    lifetimeElections:0, totalElections:0,
    electionHistory: [], // {year,type,voted,ballot,source}
  };
}

function rebuild() {
  voters = []; votersById = {}; elections = [];
  mergeInfo = { ilCount:0, enrichedCount:0, merged:0, onlyIL:0, onlyEnriched:0 };
  const precMap = {};
  const getP = (pid) => { if (!precMap[pid]) precMap[pid] = { id:pid, voters:[], lat:null, lng:null, demVotes:null, repVotes:null }; return precMap[pid]; };

  // Union of detected elections across all voter files
  const elecUnion = {};
  files.forEach(f => {
    if (f.type === 'voter_individual' || f.type === 'voter_enriched') {
      (f.elections || []).forEach(e => {
        const k = `${e.year}_${e.type}_${e.style||'x'}`;
        if (!elecUnion[k]) elecUnion[k] = e;
      });
    }
  });
  elections = Object.values(elecUnion).sort((a,b) => b.year - a.year);

  // Process IL files first (primary records)
  const ilFiles = files.filter(f => f.type === 'voter_individual');
  ilFiles.forEach(f => {
    f.rows.forEach(r => {
      const v = buildVoterFromIL(r, f);
      if (!v) return;
      // Record election history from this file
      (f.elections||[]).forEach(e => {
        const vt = e.voteTypeCol ? String(r[e.voteTypeCol] ?? '').trim() : '';
        const bc = e.ballotChoiceCol ? classifyBallot(r[e.ballotChoiceCol]) : null;
        const voted = vt !== '' && vt.toUpperCase() !== 'N' && vt.toUpperCase() !== 'NO';
        v.electionHistory.push({ year:e.year, type:e.type, voted, ballot:bc, source:'IL' });
      });
      voters.push(v); votersById[v.id] = v;
      mergeInfo.ilCount++; mergeInfo.onlyIL++;
    });
  });

  // Process enriched files — merge when ID matches, else create
  const enrichedFiles = files.filter(f => f.type === 'voter_enriched');
  enrichedFiles.forEach(f => {
    const m = f.mapping;
    f.rows.forEach(r => {
      const voterId = m.voterId ? String(r[m.voterId] ?? '').trim() : '';
      if (!voterId) return;
      mergeInfo.enrichedCount++;

      let v = votersById[voterId];
      const isNew = !v;
      if (isNew) {
        // Create from enriched only
        const firstName = m.firstName ? r[m.firstName] : '';
        const lastName = m.lastName ? r[m.lastName] : '';
        const pid = m.precinct ? String(r[m.precinct] ?? '').trim() : '';
        v = {
          id: voterId,
          name: `${firstName||''} ${lastName||''}`.trim() || `(Voter ${voterId})`,
          firstName: firstName||'', lastName: lastName||'',
          age: m.age ? num(r[m.age]) : null,
          precinctId: pid, address: m.addr1 ? r[m.addr1] : '',
          city: m.city ? r[m.city] : '', zip: m.zip ? r[m.zip] : '',
          lat: m.lat ? num(r[m.lat]) : null, lng: m.lng ? num(r[m.lng]) : null,
          sources: { il:false, enriched:true },
          phone:'', mobile:'', landline:'', dob:'', gender:'', suffix:'',
          observedParty:null, officialParty:null, calculatedParty:null,
          overallFreq:null, generalFreq:null, primaryFreq:null,
          cd:'', ld:'', sd:'', county:'', state:'',
          propensity:0, party:'U', partisanScore:0,
          dBallots:0, rBallots:0, otherBallots:0,
          votedGenerals:0, totalGenerals:0, votedPrimaries:0, totalPrimaries:0,
          last5Count:0, last5Voted:0,
          isSuperVoter:false, isBipartisanVoter:false,
          lifetimeElections:0, totalElections:0, electionHistory:[],
        };
        voters.push(v); votersById[v.id] = v;
        mergeInfo.onlyEnriched++;
      } else {
        v.sources.enriched = true;
        mergeInfo.merged++;
        mergeInfo.onlyIL--;
      }

      // Enrich
      if (m.phone) v.phone = v.phone || String(r[m.phone]||'');
      if (m.mobile) v.mobile = v.mobile || String(r[m.mobile]||'');
      if (m.landline) v.landline = v.landline || String(r[m.landline]||'');
      if (m.dob) v.dob = v.dob || String(r[m.dob]||'');
      if (m.gender) v.gender = v.gender || String(r[m.gender]||'');
      if (m.suffix) v.suffix = v.suffix || String(r[m.suffix]||'');
      if (m.observedParty) v.observedParty = normParty(r[m.observedParty]);
      if (m.officialParty) v.officialParty = normParty(r[m.officialParty]);
      if (m.calculatedParty) v.calculatedParty = normParty(r[m.calculatedParty]);
      if (m.overallFreq) v.overallFreq = num(r[m.overallFreq]);
      if (m.generalFreq) v.generalFreq = num(r[m.generalFreq]);
      if (m.primaryFreq) v.primaryFreq = num(r[m.primaryFreq]);
      if (m.cd) v.cd = String(r[m.cd]||'');
      if (m.ld) v.ld = String(r[m.ld]||'');
      if (m.sd) v.sd = String(r[m.sd]||'');
      if (m.county) v.county = String(r[m.county]||'');
      if (m.state) v.state = String(r[m.state]||'');
      if (!v.precinctId && m.precinct) v.precinctId = String(r[m.precinct]||'').trim();
      if (!v.age && m.age) v.age = num(r[m.age]);
      if (!v.address && m.addr1) v.address = String(r[m.addr1]||'');
      if (!v.city && m.city) v.city = String(r[m.city]||'');
      if (!v.zip && m.zip) v.zip = String(r[m.zip]||'');
      if (v.lat === null && m.lat) v.lat = num(r[m.lat]);
      if (v.lng === null && m.lng) v.lng = num(r[m.lng]);

      // Dem/Rep flags supplement party
      const demFlag = m.demFlag ? truthy(r[m.demFlag]) : false;
      const repFlag = m.repFlag ? truthy(r[m.repFlag]) : false;
      if (demFlag) v._flagD = true;
      if (repFlag) v._flagR = true;

      // VH elections — only add if not already in history (prevent double-count)
      (f.elections||[]).forEach(e => {
        const existing = v.electionHistory.find(h => h.year === e.year && h.type === e.type);
        const vt = e.voteTypeCol ? String(r[e.voteTypeCol] ?? '').trim() : '';
        const voted = vt !== '' && vt.toUpperCase() !== 'N' && vt.toUpperCase() !== 'NO';
        if (!existing) {
          v.electionHistory.push({ year:e.year, type:e.type, voted, ballot:null, source:'VH', raw:vt });
        } else if (!existing.voted && voted) {
          // VH confirms turnout IL file missed
          existing.voted = true;
        }
      });
    });
  });

  // Now compute scores for every voter using their unified electionHistory
  voters.forEach(v => {
    const hist = v.electionHistory;
    let votedGen = 0, totGen = 0, votedPri = 0, totPri = 0;
    let dB = 0, rB = 0, oB = 0;
    hist.forEach(h => {
      if (h.type === 'general') { totGen++; if (h.voted) votedGen++; }
      if (h.type === 'primary') {
        totPri++;
        if (h.voted) {
          votedPri++;
          if (h.ballot === 'D') dB++;
          else if (h.ballot === 'R') rB++;
          else if (h.ballot) oB++;
        }
      }
    });
    if (v._flagD) dB = Math.max(dB, 1);
    if (v._flagR) rB = Math.max(rB, 1);
    delete v._flagD; delete v._flagR;

    const sorted = [...hist].sort((a,b) => (b.year||0) - (a.year||0));
    const last5 = sorted.slice(0,5);
    const weights = [0.30,0.25,0.20,0.15,0.10].slice(0, last5.length);
    let wSum = 0, wTot = 0;
    last5.forEach((h,i) => { if (weights[i]) { wSum += (h.voted?1:0)*weights[i]; wTot += weights[i]; } });
    let propensity = wTot ? (wSum/wTot)*100 : 0;
    // If we have frequency scores from enriched file, average in
    if (v.overallFreq !== null && v.overallFreq !== undefined) {
      const f = v.overallFreq <= 1 ? v.overallFreq*100 : v.overallFreq;
      propensity = (propensity*0.6 + f*0.4);
    }

    let party = 'U', partisanScore = 0;
    if (dB + rB > 0) {
      partisanScore = ((dB - rB)/(dB + rB))*100;
      if (dB > rB*1.5) party = 'D';
      else if (rB > dB*1.5) party = 'R';
      else party = 'U';
    }
    // If enriched file provides a clear party, honor it (priority: official > calculated > observed)
    const enrichedParty = v.officialParty || v.calculatedParty || v.observedParty;
    if (enrichedParty) {
      party = enrichedParty;
      // override partisan score signal when no ballot history exists
      if (dB + rB === 0) partisanScore = enrichedParty === 'D' ? 70 : enrichedParty === 'R' ? -70 : 0;
    }

    v.propensity = +propensity.toFixed(1);
    v.party = party;
    v.partisanScore = +partisanScore.toFixed(1);
    v.dBallots = dB; v.rBallots = rB; v.otherBallots = oB;
    v.votedGenerals = votedGen; v.totalGenerals = totGen;
    v.votedPrimaries = votedPri; v.totalPrimaries = totPri;
    v.last5Count = last5.length;
    v.last5Voted = last5.filter(h => h.voted).length;
    v.isSuperVoter = v.last5Count >= 4 && v.last5Voted >= 4;
    v.isBipartisanVoter = (dB >= 1 && rB >= 1);
    v.lifetimeElections = hist.filter(h => h.voted).length;
    v.totalElections = hist.length;

    // attach voter to precinct map
    const p = getP(v.precinctId || '(unassigned)');
    p.voters.push(v);
    if (v.lat !== null && v.lng !== null && p.lat === null) { p.lat = v.lat; p.lng = v.lng; }
  });

  // Legacy/precinct-level files
  files.forEach(f => {
    if (f.type === 'voter_individual' || f.type === 'voter_enriched') return;
    const m = f.mapping;
    f.rows.forEach(r => {
      const pid = m.precinct ? String(r[m.precinct] ?? '').trim() : '';
      if (!pid) return;
      const p = getP(pid);
      if (f.type === 'results') {
        const dv = num(r[m.demVotes]); if (dv !== null) p.demVotes = (p.demVotes || 0) + dv;
        const rv = num(r[m.repVotes]); if (rv !== null) p.repVotes = (p.repVotes || 0) + rv;
      } else if (f.type === 'geo') {
        const la = num(r[m.lat]), lo = num(r[m.lng]);
        if (la !== null && lo !== null) { p.lat = la; p.lng = lo; }
      } else if (f.type === 'voter' && !voters.length) {
        const reg = num(r[m.registered]) || 0;
        p._simpleReg = (p._simpleReg||0) + reg;
        p._simpleDem = num(r[m.dem]) || 0;
        p._simpleRep = num(r[m.rep]) || 0;
        p._simpleTurnout = num(r[m.turnout]) || null;
      }
    });
  });

  // Build precinct summary
  precincts = Object.values(precMap).map(p => {
    let reg = 0, demCount = 0, repCount = 0, unaffCount = 0;
    let propSum = 0, propN = 0, superCount = 0, biCount = 0;
    let histLean = 0;
    if (p.voters.length) {
      reg = p.voters.length;
      p.voters.forEach(v => {
        if (v.party === 'D') demCount++; else if (v.party === 'R') repCount++; else unaffCount++;
        if (v.propensity !== null) { propSum += v.propensity; propN++; }
        if (v.isSuperVoter) superCount++;
        if (v.isBipartisanVoter) biCount++;
      });
      histLean = p.voters.reduce((s,v)=>s+v.partisanScore,0) / p.voters.length;
    } else if (p._simpleReg) {
      reg = p._simpleReg;
      demCount = Math.round(reg * (p._simpleDem||0)/100);
      repCount = Math.round(reg * (p._simpleRep||0)/100);
      unaffCount = Math.max(0, reg - demCount - repCount);
      histLean = (p._simpleDem||0) - (p._simpleRep||0);
    }
    const demPct = reg ? (demCount/reg)*100 : 0;
    const repPct = reg ? (repCount/reg)*100 : 0;
    const unaffPct = reg ? (unaffCount/reg)*100 : 0;
    let propensity = propN ? propSum/propN : (p._simpleTurnout || 50);
    let lean = histLean;
    if (p.demVotes && p.repVotes) {
      const tot = p.demVotes + p.repVotes;
      lean = lean*0.6 + ((p.demVotes - p.repVotes)/tot*100)*0.4;
    }
    const absLean = Math.abs(lean);
    const persuasion = Math.max(0, Math.min(100, (100 - absLean*2)*0.5 + unaffPct*0.8));
    const superPct = reg ? (superCount/reg)*100 : (propensity>=70?60:propensity>=60?40:20);
    let tier;
    if (absLean > 10 && propensity < 60) tier = 1;
    else if (absLean <= 10) tier = 2;
    else if (absLean > 10 && propensity >= 60) tier = 3;
    else tier = 4;
    const score = Math.max(0, Math.min(100, Math.round((Math.max(0,lean)*1.5) + (lean>5&&propensity<65?30:0) + (absLean<10?20:0) + reg/100)));
    return {
      id: p.id, reg, dem: demCount, rep: repCount, unaff: unaffCount,
      demPct: +demPct.toFixed(1), repPct: +repPct.toFixed(1), unaffPct: +unaffPct.toFixed(1),
      propensity: +propensity.toFixed(1), superPct: +superPct.toFixed(1),
      persuasion: +persuasion.toFixed(1), lean: +lean.toFixed(1),
      score, tier, lat: p.lat, lng: p.lng,
      demVotes: p.demVotes, repVotes: p.repVotes, biCount,
      hasVoterData: p.voters.length > 0,
    };
  });
}

// ================ SCORING ================
function targetScore(p) {
  const leanForUs = ourSide === 'D' ? p.lean : -p.lean;
  const leanNorm = Math.max(0, Math.min(100, leanForUs + 50));
  const volNorm = Math.min(100, p.reg / 10);
  const total = scoreWeights.propensity + scoreWeights.lean + scoreWeights.persuasion + scoreWeights.volume || 1;
  return +((p.propensity*scoreWeights.propensity + leanNorm*scoreWeights.lean + p.persuasion*scoreWeights.persuasion + volNorm*scoreWeights.volume)/total).toFixed(1);
}
function isSuper(p) { return p.superPct >= 40 || p.propensity >= 75; }
function isBipartisan(p) { return Math.abs(p.lean) <= 10 && (p.unaffPct >= 25 || p.persuasion >= 50 || p.biCount >= p.reg*0.1); }

function applyVoterFilters() {
  const q = document.getElementById('voterSearch').value.toLowerCase();
  const party = document.getElementById('voterPartyFilter').value;
  const seg = document.getElementById('voterSegFilter').value;
  let list = voters;
  if (q) list = list.filter(v => (v.name||'').toLowerCase().includes(q) || (v.id||'').toLowerCase().includes(q) || (v.phone||'').toLowerCase().includes(q) || (v.mobile||'').toLowerCase().includes(q));
  if (party) list = list.filter(v => v.party === party);
  if (seg === 'super') list = list.filter(v => v.isSuperVoter);
  else if (seg === 'bi') list = list.filter(v => v.isBipartisanVoter || v.party === 'U');
  else if (seg === 'prop') list = list.filter(v => v.propensity >= 70);
  else if (seg === 'low') list = list.filter(v => v.propensity < 40);
  else if (seg === 'merged') list = list.filter(v => v.sources.il && v.sources.enriched);
  else if (seg === 'hasphone') list = list.filter(v => v.phone || v.mobile);
  const tb = document.getElementById('voterTbody');
  tb.innerHTML = list.slice(0, 500).map(v => {
    const tags = [];
    if (v.sources.il && v.sources.enriched) tags.push('<span class="badge-merged px-1 py-0.5 rounded text-xs">🔗</span>');
    if (v.isSuperVoter) tags.push('<span class="badge-super px-1 py-0.5 rounded text-xs">⭐</span>');
    if (v.isBipartisanVoter) tags.push('<span class="badge-bi px-1 py-0.5 rounded text-xs">🔀</span>');
    if (v.propensity >= 70) tags.push('<span class="badge-prop px-1 py-0.5 rounded text-xs">📈</span>');
    const phoneDisp = v.mobile || v.phone || '';
    return `<tr class="border-b hover:bg-slate-50">
      <td class="p-2 font-mono text-xs">${(v.id||'').slice(0,14)}</td>
      <td class="p-2"><b>${v.name}</b>${v.gender?'<span class="text-slate-400"> · '+v.gender+'</span>':''}</td>
      <td class="p-2 text-right">${v.age ?? ''}</td>
      <td class="p-2 text-xs">${phoneDisp}</td>
      <td class="p-2 text-xs">${v.precinctId||''}</td>
      <td class="p-2 text-center"><span class="badge-${v.party.toLowerCase()} px-2 py-0.5 rounded text-xs">${v.party}</span></td>

  }).join('') || `<tr><td colspan="10" class="p-4 text-center text-slate-500">No voters match.</td></tr>`;
}

function getFilteredVoters() {
  const pMin = +document.getElementById('fPropMin').value;
  const pMax = +document.getElementById('fPropMax').value;
  const lMin = +document.getElementById('fLeanMin').value;
  const lMax = +document.getElementById('fLeanMax').value;
  const aMin = +document.getElementById('fMinAge').value;
  const aMax = +document.getElementById('fMaxAge').value;
  const su = document.getElementById('fSuperOnly').checked;
  const bi = document.getElementById('fBipartisan').checked;
  const ph = document.getElementById('fHasPhone').checked;
  return voters.filter(v => {
    const leanForUs = ourSide==='D' ? v.partisanScore : -v.partisanScore;
    if (v.propensity < pMin || v.propensity > pMax) return false;
    if (leanForUs < lMin || leanForUs > lMax) return false;
    if (v.age !== null && v.age !== undefined && v.age !== '' && (v.age < aMin || v.age > aMax)) return false;
    if (su && !v.isSuperVoter) return false;
    if (bi && !(v.isBipartisanVoter || v.party==='U')) return false;
    if (ph && !(v.phone || v.mobile)) return false;
    return true;
  });
}
function getFilteredPrecincts() {
  const pMin = +document.getElementById('fPropMin').value;
  const pMax = +document.getElementById('fPropMax').value;
  const lMin = +document.getElementById('fLeanMin').value;
  const lMax = +document.getElementById('fLeanMax').value;
  const su = document.getElementById('fSuperOnly').checked;
  const bi = document.getElementById('fBipartisan').checked;
  const tiers = [...document.querySelectorAll('.tier-chk')].filter(c => c.checked).map(c => +c.value);
  return precincts.filter(p => {
    const leanForUs = ourSide==='D' ? p.lean : -p.lean;
    if (p.propensity < pMin || p.propensity > pMax) return false;
    if (leanForUs < lMin || leanForUs > lMax) return false;
    if (su && !isSuper(p)) return false;
    if (bi && !isBipartisan(p)) return false;
    if (!tiers.includes(p.tier)) return false;
    return true;
  }).map(p => ({...p, tgScore: targetScore(p)}));
}

function updateTargeting() {
  if (!precincts.length) return;
  if (voters.length) {
    document.getElementById('count-super').textContent = voters.filter(v=>v.isSuperVoter).length.toLocaleString();
    document.getElementById('count-bi').textContent = voters.filter(v=>v.isBipartisanVoter||v.party==='U').length.toLocaleString();
    document.getElementById('count-prop').textContent = voters.filter(v=>v.propensity>=70).length.toLocaleString();
    document.getElementById('count-low').textContent = voters.filter(v => {
      const l = ourSide==='D' ? v.partisanScore : -v.partisanScore;
      return v.propensity<40 && l>0;
    }).length.toLocaleString();
  } else {
    document.getElementById('count-super').textContent = precincts.filter(isSuper).length;
    document.getElementById('count-bi').textContent = precincts.filter(isBipartisan).length;
    document.getElementById('count-prop').textContent = precincts.filter(p=>p.propensity>=70).length;
    document.getElementById('count-low').textContent = precincts.filter(p => { const l=ourSide==='D'?p.lean:-p.lean; return p.propensity<40 && l>0; }).length;
  }
  const filteredPrecincts = getFilteredPrecincts();
  const filteredVoters = voters.length ? getFilteredVoters() : [];
  const sortKey = document.getElementById('tgSort').value;
  const sortFn = { score:(a,b)=>b.tgScore-a.tgScore, propensity:(a,b)=>b.propensity-a.propensity, persuasion:(a,b)=>b.persuasion-a.persuasion, lean:(a,b)=>ourSide==='D'?b.lean-a.lean:a.lean-b.lean, reg:(a,b)=>b.reg-a.reg }[sortKey];
  filteredPrecincts.sort(sortFn);
  const totReg = filteredPrecincts.reduce((s,p)=>s+p.reg,0);
  const projOurs = Math.round(filteredPrecincts.reduce((s,p)=>s+(ourSide==='D'?p.dem:p.rep)*(p.propensity/100),0));
  document.getElementById('tgVoterCount').textContent = (voters.length?filteredVoters.length:totReg).toLocaleString();

    return `<tr class="border-b hover:bg-slate-50">
      <td class="p-2"><b>${p.id}</b></td>
      <td class="p-2 text-right font-bold text-blue-700">${p.tgScore}</td>
      <td class="p-2 text-right">${p.propensity}</td>
      <td class="p-2 text-right">${p.persuasion}</td>
      <td class="p-2 text-right ${l>=0?'text-blue-700':'text-red-600'}">${l>=0?'+':''}${l.toFixed(1)}</td>
      <td class="p-2 text-right">${p.reg.toLocaleString()}</td>
      <td class="p-2 text-center">${tags.join(' ')||'—'}</td>
    </tr>`;
  }).join('') || `<tr><td colspan="7" class="p-4 text-center text-slate-500">No precincts match.</td></tr>`;
  const buckets = [0,0,0,0,0];
  filteredPrecincts.forEach(p => buckets[Math.min(4, Math.floor(p.tgScore/20))]++);
  if (window._tgc) window._tgc.destroy();
  window._tgc = new Chart(document.getElementById('tgDistChart'), {
    type:'bar',
    data:{labels:['0-20','20-40','40-60','60-80','80-100'],datasets:[{label:'Precincts',data:buckets,backgroundColor:['#cbd5e1','#94a3b8','#60a5fa','#3b82f6','#1d4ed8']}]},

  });

let mapState = { mode:'priority', tx:0, ty:0, scale:1, dragging:false, sx:0, sy:0 };
const mapSvg = document.getElementById('realmap');
function renderMap() {
  const withCoords = precincts.filter(p => p.lat !== null && p.lng !== null);
  const status = document.getElementById('mapStatus');
  if (!withCoords.length) {
    status.textContent = '⚠ No precinct coordinates. Upload a geo file.';
    mapSvg.innerHTML = `<text x="50%" y="50%" text-anchor="middle" fill="#64748b" font-size="14">No coordinates available</text>`;
    document.getElementById('mapLegend').innerHTML = ''; return;
  }
  status.textContent = `${withCoords.length} of ${precincts.length} precincts have coordinates.`;
  const lats = withCoords.map(p=>p.lat), lngs = withCoords.map(p=>p.lng);
  const mnL = Math.min(...lats), mxL = Math.max(...lats), mnLg = Math.min(...lngs), mxLg = Math.max(...lngs);
  const pL = (mxL-mnL)*0.1 || 0.01, pLg = (mxLg-mnLg)*0.1 || 0.01;
  mapState.bounds = { minLat:mnL-pL, maxLat:mxL+pL, minLng:mnLg-pLg, maxLng:mxLg+pLg };
  drawMap(); renderMapLegend();
}
function drawMap() {
  const rect = mapSvg.getBoundingClientRect();
  const W = rect.width, H = rect.height, b = mapState.bounds;
  if (!b) return;
  const withCoords = precincts.filter(p => p.lat !== null && p.lng !== null);
  mapSvg.innerHTML = '';
  mapSvg.setAttribute('viewBox', `0 0 ${W} ${H}`);
  const g = document.createElementNS('http://www.w3.org/2000/svg','g');
  g.setAttribute('transform', `translate(${mapState.tx},${mapState.ty}) scale(${mapState.scale})`);
  const pad = 30;
  withCoords.forEach(p => {
    const x = pad + ((p.lng-b.minLng)/(b.maxLng-b.minLng))*(W-2*pad);
    const y = H - pad - ((p.lat-b.minLat)/(b.maxLat-b.minLat))*(H-2*pad);
    const r = Math.max(6, Math.min(20, Math.sqrt(p.reg/5)));
    const c = document.createElementNS('http://www.w3.org/2000/svg','circle');
    c.setAttribute('cx',x); c.setAttribute('cy',y); c.setAttribute('r',r);
    c.setAttribute('fill',colorFor(p, mapState.mode)); c.setAttribute('fill-opacity','0.75');
    c.setAttribute('stroke','#fff'); c.setAttribute('stroke-width','1.5'); c.setAttribute('class','marker');
    c.addEventListener('mouseenter', e => showTooltip(e,p));
    c.addEventListener('mousemove', e => showTooltip(e,p));
    c.addEventListener('mouseleave', hideTooltip);
    c.addEventListener('click', () => alert(`${p.id}\nVoters: ${p.reg}\nD ${p.demPct}% / R ${p.repPct}% / U ${p.unaffPct}%\nProp: ${p.propensity} · Super ${p.superPct}%\nTarget: ${targetScore(p)}\nTier ${p.tier}`));
    g.appendChild(c);
  });
  mapSvg.appendChild(g);
}
function colorFor(p, m) {
  if (m==='priority') return ['#dc2626','#ea580c','#ca8a04','#16a34a'][p.tier-1];
  if (m==='partisan') { const l = ourSide==='D'?p.lean:-p.lean; if (l>=15)return'#1e3a8a';if (l>=5)return'#3b82f6';if (l>=-5)return'#a78bfa';if (l>=-15)return'#f87171';return'#991b1b'; }
  if (m==='propensity') { if (p.propensity>=75)return'#166534';if (p.propensity>=60)return'#22c55e';if (p.propensity>=45)return'#eab308';return'#ef4444'; }
  if (m==='target') { const s=targetScore(p); if (s>=75)return'#1e3a8a';if (s>=60)return'#3b82f6';if (s>=45)return'#60a5fa';if (s>=30)return'#cbd5e1';return'#94a3b8'; }
  return '#64748b';
}
function renderMapLegend() {
  const l = document.getElementById('mapLegend');
  const items = {
    priority:[['#dc2626','T1'],['#ea580c','T2'],['#ca8a04','T3'],['#16a34a','T4']],
    partisan:[['#1e3a8a','Strong Us'],['#3b82f6','Lean Us'],['#a78bfa','Swing'],['#f87171','Lean Opp'],['#991b1b','Strong Opp']],
    propensity:[['#166534','75+'],['#22c55e','60-74'],['#eab308','45-59'],['#ef4444','<45']],
    target:[['#1e3a8a','75+'],['#3b82f6','60-74'],['#60a5fa','45-59'],['#cbd5e1','30-44'],['#94a3b8','<30']]
  };
  l.innerHTML = items[mapState.mode].map(([c,t]) => `<span class="px-2 py-0.5 rounded text-white" style="background:${c}">${t}</span>`).join(' ');
}
function showTooltip(ev, p) {
  const t = document.getElementById('mapTooltip'); const box = mapSvg.getBoundingClientRect();
  t.classList.remove('hidden');
  t.innerHTML = `<b>${p.id}</b><br>${p.reg} voters<br>D${p.demPct}% R${p.repPct}% U${p.unaffPct}%<br>Prop ${p.propensity} · Super ${p.superPct}%<br>Target ${targetScore(p)}`;
  t.style.left = (ev.clientX - box.left + 10) + 'px';
  t.style.top = (ev.clientY - box.top + 10) + 'px';
}
function hideTooltip() { document.getElementById('mapTooltip').classList.add('hidden'); }
mapSvg.addEventListener('mousedown', e => { mapState.dragging=true; mapState.sx=e.clientX-mapState.tx; mapState.sy=e.clientY-mapState.ty; });

function initWinCalc() {
  const reg = voters.length || precincts.reduce((s,p)=>s+p.reg,0);
  const avgP = voters.length ? voters.reduce((s,v)=>s+v.propensity,0)/voters.length : (precincts.length?precincts.reduce((s,p)=>s+p.propensity,0)/precincts.length:60);
  document.getElementById('wnReg').value = reg;
  document.getElementById('wnTurnout').value = avgP.toFixed(1);
  calcWin();
}
function calcWin() {
  const r=+document.getElementById('wnReg').value, t=+document.getElementById('wnTurnout').value/100;
  const w=+document.getElementById('wnWinPct').value/100, pr=+document.getElementById('wnPersuade').value/100;
  const tot=Math.round(r*t), win=Math.round(tot*w), pn=Math.round(win*pr);
  document.getElementById('calcTotal').textContent=tot.toLocaleString();
  document.getElementById('calcWin').textContent=win.toLocaleString();
  document.getElementById('calcBase').textContent=(win-pn).toLocaleString();
  document.getElementById('calcPersuade').textContent=pn.toLocaleString();
}
